import os
from PIL import Image

def convert_and_resize_images(folder_path):
    supported_formats = ['.jpg', '.jpeg', '.bmp', '.gif', '.tiff', '.webp', '.png']

    for filename in os.listdir(folder_path):
        file_path = os.path.join(folder_path, filename)
        name, ext = os.path.splitext(filename)

        if ext.lower() not in supported_formats:
            continue

        try:
            with Image.open(file_path) as img:
                # Convert to PNG if not already
                if ext.lower() != '.png':
                    output_path = os.path.join(folder_path, f"{name}.png")
                    img.save(output_path, "PNG")
                    print(f"Converted {filename} to PNG")
                else:
                    output_path = file_path

                # Resize if width is not 200
                with Image.open(output_path) as png_img:
                    width, height = png_img.size
                    if width != 200:
                        new_height = int((200 / width) * height)
                        resized_img = png_img.resize((200, new_height), Image.LANCZOS)
                        resized_img.save(output_path)
                        print(f"Resized {os.path.basename(output_path)} to 200px width")
        except Exception as e:
            print(f"Failed to process {filename}: {e}")

if __name__ == "__main__":
    current_folder = os.getcwd()  # Use the folder where the script is run
    convert_and_resize_images(current_folder)
